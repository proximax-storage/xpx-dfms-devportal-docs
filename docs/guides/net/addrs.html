Get Node Addresses

Requirements

  • IDE or text editor
  • Have one owner or replicator node

Example

//Some enviroment
var env cmds.Environment

//New network api
api, err := APINetwork(env)
if err != nil {
    return err
}

//Get addresses
addrs, err := api.Addrs(context.Background())
if err != nil {
    return err
}

//Print addresses
for _, addr := range addrs {
    println(addr.String())
}
